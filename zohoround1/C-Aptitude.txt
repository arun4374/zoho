#include <iostream>

int main() {
    int c =3,no = 30;
    do{
        no /= c;     
    }while(!c=0);

    printf("%d/n",no);

    return 0;
}

// output: 10->3->1.....n      (infinity loop because while contition only check !c=0 but c value not reduced so c is same value(3) so, this is infinity loop)
